Table categories {
  category_id int [pk, increment]
  name varchar(100) [not null]
}

Table manufacturers {
  manufacturer_id int [pk, increment]
  name varchar(150) [not null]
  country varchar(100)
  contact varchar(100)
}

Table suppliers {
  supplier_id int [pk, increment]
  company_name varchar(150) [not null]
  contact_person varchar(100)
  phone varchar(20)
  email varchar(100)
  city varchar(100)
  is_active boolean [default: true]
}

Table medicines {
  medicine_id int [pk, increment]
  category_id int [ref: > categories.category_id]
  manufacturer_id int [ref: > manufacturers.manufacturer_id]
  supplier_id int [ref: > suppliers.supplier_id]
  name varchar(150) [not null]
  dosage_form varchar(50)
  unit_price decimal(10,2) [not null]
  expiry_date date [not null]
  batch_no varchar(50)
}

Table stock {
  stock_id int [pk, increment]
  medicine_id int [ref: - medicines.medicine_id]
  quantity int [default: 0]
  low_stock_threshold int [default: 20]
  last_updated datetime
}

Table stock_transactions {
  transaction_id int [pk, increment]
  medicine_id int [ref: > medicines.medicine_id]
  type varchar(20) [note: 'IN or OUT']
  quantity int [not null]
  reason varchar(255)
  created_at datetime
}

Table purchase_orders {
  order_id int [pk, increment]
  supplier_id int [ref: > suppliers.supplier_id]
  order_date date [not null]
  expected_delivery date
  status varchar(20) [note: 'Pending, Delivered, Cancelled']
}

Table purchase_order_items {
  item_id int [pk, increment]
  order_id int [ref: > purchase_orders.order_id]
  medicine_id int [ref: > medicines.medicine_id]
  quantity int [not null]
  unit_price decimal(10,2)
}

Table doctors {
  doctor_id int [pk, increment]
  name varchar(100) [not null]
  specialization varchar(100)
  phone varchar(20)
}

Table patients {
  patient_id int [pk, increment]
  name varchar(100) [not null]
  phone varchar(20)
  dob date
  address varchar(255)
}

Table prescriptions {
  prescription_id int [pk, increment]
  patient_id int [ref: > patients.patient_id]
  doctor_id int [ref: > doctors.doctor_id]
  prescribed_date date [not null]
  notes text
}

Table prescription_items {
  item_id int [pk, increment]
  prescription_id int [ref: > prescriptions.prescription_id]
  medicine_id int [ref: > medicines.medicine_id]
  quantity int [not null]
  instructions text
}

Table dispense_records {
  dispense_id int [pk, increment]
  prescription_id int [ref: > prescriptions.prescription_id]
  medicine_id int [ref: > medicines.medicine_id]
  quantity int [not null]
  dispensed_at datetime
  dispensed_by varchar(100)
}
