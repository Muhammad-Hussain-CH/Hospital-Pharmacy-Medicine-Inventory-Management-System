-- Q17: Total stock value of entire pharmacy inventory
-- Purpose: Financial summary of stock on hand
SELECT 
    SUM(m.unit_price * s.quantity) AS total_inventory_value,
    COUNT(m.medicine_id)           AS total_medicines,
    SUM(s.quantity)                AS total_units_in_stock
FROM medicines m
JOIN stock s ON m.medicine_id = s.medicine_id
WHERE m.expiry_date >= CURDATE();

-- Q18: Top 5 most dispensed medicines
-- Purpose: Identifies high-demand medicines for procurement
SELECT 
    m.name              AS medicine_name,
    m.dosage_form,
    SUM(dr.quantity)    AS total_dispensed,
    COUNT(dr.dispense_id) AS dispense_count
FROM dispense_records dr
JOIN medicines m ON dr.medicine_id = m.medicine_id
GROUP BY m.medicine_id, m.name, m.dosage_form
ORDER BY total_dispensed DESC
LIMIT 5;

-- Q19: Total medicines dispensed per doctor
-- Purpose: Doctor activity report
SELECT 
    d.name              AS doctor_name,
    d.specialization,
    COUNT(rx.prescription_id) AS total_prescriptions,
    SUM(dr.quantity)          AS total_medicines_dispensed
FROM doctors d
JOIN prescriptions  rx ON d.doctor_id        = rx.doctor_id
JOIN dispense_records dr ON rx.prescription_id = dr.prescription_id
GROUP BY d.doctor_id, d.name, d.specialization
ORDER BY total_prescriptions DESC;

-- Q20: Monthly dispense report for current month
-- Purpose: Monthly activity summary for management
SELECT 
    m.name              AS medicine_name,
    SUM(dr.quantity)    AS total_dispensed,
    COUNT(dr.dispense_id) AS number_of_times,
    SUM(dr.quantity * m.unit_price) AS total_value
FROM dispense_records dr
JOIN medicines m ON dr.medicine_id = m.medicine_id
WHERE MONTH(dr.dispensed_at) = MONTH(CURDATE())
  AND YEAR(dr.dispensed_at)  = YEAR(CURDATE())
GROUP BY m.medicine_id, m.name
ORDER BY total_dispensed DESC;

-- Q21: Total order value per supplier
-- Purpose: Supplier spending analysis
SELECT 
    s.company_name      AS supplier,
    COUNT(po.order_id)  AS total_orders,
    SUM(poi.quantity * poi.unit_price) AS total_order_value
FROM suppliers s
JOIN purchase_orders     po  ON s.supplier_id  = po.supplier_id
JOIN purchase_order_items poi ON po.order_id   = poi.order_id
WHERE po.status != 'Cancelled'
GROUP BY s.supplier_id, s.company_name
ORDER BY total_order_value DESC;
